import * as fs from "node:fs";
import * as path from "node:path";

/**
 * Extract PascalCase glossary terms from an "## Ubiquitous Language" section.
 * Handles: **Term**, - Term:, and | Term | table formats.
 */
export function extractGlossaryTerms(content: string): Set<string> {
  const terms = new Set<string>();
  // Extract the Ubiquitous Language section (up to next ## heading)
  const sectionMatch = content.match(
    /## Ubiquitous Language([\s\S]*?)(?:\n## |\z)/
  );
  const section = sectionMatch ? sectionMatch[1] : content;

  // Match **PascalCase** bold format
  for (const m of section.matchAll(/\*\*([A-Z][a-zA-Z0-9]+)\*\*/g)) {
    terms.add(m[1]);
  }
  // Match - PascalCase: list format
  for (const m of section.matchAll(/^[-*]\s+([A-Z][a-zA-Z0-9]+)\s*:/gm)) {
    terms.add(m[1]);
  }
  // Match | PascalCase | table format
  for (const m of section.matchAll(/\|\s*([A-Z][a-zA-Z0-9]+)\s*\|/g)) {
    terms.add(m[1]);
  }

  return terms;
}

/**
 * Returns terms that appear in the candidate list but are not in the glossary.
 */
export function validateDomainGlossary(
  glossaryContent: string,
  candidateTerms: string[]
): string[] {
  if (candidateTerms.length === 0) return [];
  const known = extractGlossaryTerms(glossaryContent);
  return candidateTerms.filter((t) => !known.has(t));
}

/**
 * Load DOMAIN-MODEL.md (preferred) or DOMAIN.md from the planning directory.
 * Returns the file content, or empty string if neither exists.
 */
export function loadDomainContent(cwd: string): string {
  const planningDir = path.join(cwd, ".planning");
  const modelPath = path.join(planningDir, "DOMAIN-MODEL.md");
  if (fs.existsSync(modelPath)) {
    return fs.readFileSync(modelPath, "utf-8");
  }
  const domainPath = path.join(planningDir, "DOMAIN.md");
  if (fs.existsSync(domainPath)) {
    return fs.readFileSync(domainPath, "utf-8");
  }
  return "";
}

function extractGlossaryTerms(content) {
  const terms = new Set();
  const sectionMatch = content.match(/## Ubiquitous Language([\s\S]*?)(?:\n## |$)/);
  const section = sectionMatch ? sectionMatch[1] : content;
  for (const m of section.matchAll(/\*\*([A-Z][a-zA-Z0-9]+)\*\*/g)) terms.add(m[1]);
  for (const m of section.matchAll(/^[-*]\s+([A-Z][a-zA-Z0-9]+)\s*:/gm)) terms.add(m[1]);

  return terms;
}

// 5. Domain glossary compliance (checks DOMAIN-MODEL.md, falls back to DOMAIN.md)
const domainContent = loadDomainContent(cwd);
if (domainContent) {
  try {
    const glossaryTerms = extractGlossaryTerms(domainContent);
    const changedFiles = execSync(
      "git diff --cached --name-only 2>/dev/null || git diff --name-only HEAD~1",
      { encoding: "utf-8" }
    ).trim().split("\n").filter((f) => f.endsWith(".ts") || f.endsWith(".tsx"));

    const unknownTerms = [];
    for (const file of changedFiles) {
      if (!fs.existsSync(file)) continue;
      const src = fs.readFileSync(file, "utf-8");
      const declarations = [...src.matchAll(/(?:class|interface|type|enum)\s+([A-Z][a-zA-Z0-9]+)/g)].map((m) => m[1]);
      for (const term of declarations) {
        if (!glossaryTerms.has(term)) unknownTerms.push(`${file}: ${term}`);
      }
    }
    if (unknownTerms.length > 0) {
      issues.push({
        severity: hookConfig.tddMode === "strict" ? "error" : "warning",
        message: `${unknownTerms.length} PascalCase type(s) not in domain glossary (DOMAIN-MODEL.md)`,
        details: unknownTerms.slice(0, 5).join(", "),
      });
    }
  } catch { /* ignore */ }
